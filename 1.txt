CREATE TABLE orders (
  id SERIAL PRIMARY KEY,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  contractor TEXT,
  sellingPoint TEXT,
  type TEXT,
  status TEXT
)

CREATE TABLE orderContent (
  id SERIAL PRIMARY KEY,
  order_id INT NOT NULL,
  code TEXT NOT NULL,
  name TEXT,
  type TEXT,
  quant TEXT,
  ordered TEXT,
  shipped TEXT
  content TEXT NOT NULL,
  CONSTRAINT fk_order FOREIGN KEY(order_id) REFERENCES orders(id)
)
